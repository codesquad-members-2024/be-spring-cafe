
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(255) UNIQUE,
    user_password VARCHAR(255),
    user_nickname VARCHAR(255),
    user_email VARCHAR(255)
);


CREATE TABLE articles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(255),
    title VARCHAR(255),
    content TEXT,
    time TIMESTAMP,
    views INT DEFAULT 0,
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);
